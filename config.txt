cp -a /etc/haproxy/haproxy.cfg{,.bak}
cat <<EOF >> vi /etc/haproxy/haproxy.cfg
frontend haproxy-main
    bind *:80
    option forwardfor
    default_backend apache_webservers

backend apache_webservers
    balance roundrobin
    server websvr1      10.10.1.30:80 check
    server websvr2      10.10.1.31:80 check


listen stats
    bind :8800
    stats enable
    stats uri /
    stats hide-version
    stats auth hamid:hamid
    default_backend apache_webservers
EOF
